  styleUrls: ['./alphabets.page.scss'],
  standalone: true,
  imports: [IonicModule, CommonModule, FormsModule],

  speak = async (data: string) => {

}

    await TextToSpeech.speak({
      text: toSpeak,

      rate: 1.0,
      pitch: 0.5,
      volume: 1.0,

    });
  };

  };
